Notes:

Rule order matters; first matching rule is applied.
Rows not matching any rule use default keys.

```yaml
universes:
  MARS_SUBSET:
    description: subset created manually
    filters: {}
    peer_group:
      keys: [asset_class, ccy, rf_level1]   # default peer grouping
      rules:
        - id: cp_rate_usd_corp
          when:
            asset_class: {eq: CP_RATE}
            ccy: {in: [USD]}
            rf_level1: {eq: CP_RATE}
            rf_level3: {like: "CORP%"}
          keys: [asset_class, ccy, rf_level1, rf_level2, rf_level3]

        - id: cp_rate_usd_rmbs
          when:
            asset_class: {eq: CP_RATE}
            ccy: {eq: USD}
            rf_level3: {like: "RMBS%"}
            rf_level4: {not_like: "TEST%"}
          keys: [asset_class, ccy, rf_level1, rf_level3, rf_level4, rf_level5]

        - id: ir_rate_usd_treasury
          when:
            asset_class: {eq: IR_RATE}
            ccy: {eq: USD}
            rf_level3: {in: [Treasury, UST]}
          keys: [asset_class, ccy, rf_level1, rf_level3, rf_level5]

        - id: eq_vol_usd_atm
          when:
            asset_class: {eq: EQ_VOL}
            ccy: {eq: USD}
            rf_level4: {like: "ATM%"}
          keys: [asset_class, ccy, rf_level1, rf_level3, rf_level4, rf_level5]

        - id: fx_vol_g10
          when:
            asset_class: {eq: FX_VOL}
            ccy: {in: [USD, EUR, GBP, JPY, CHF, AUD, CAD, NZD, NOK, SEK]}
          keys: [asset_class, ccy, rf_level1, rf_level3, rf_level5]
    use_common_checks: true
    checks: []
```
